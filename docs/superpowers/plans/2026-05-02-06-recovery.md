```python
def init_recovery_tasks(recruit_id: int, db: Session):
    """为已封门的考场创建恢复任务"""
    rcs = db.query(RecruitmentClassroom).filter(
        RecruitmentClassroom.recruitment_id == recruit_id
    ).all()
    if not rcs:
        return

    rc_ids = [rc.id for rc in rcs]
    db.query(TaskProgress).filter(
        TaskProgress.recruitment_classroom_id.in_(rc_ids),
        TaskProgress.task_type == "recovery"
    ).delete(synchronize_session=False)

    for rc in rcs:
        for item_key, item_name, _ in RECOVERY_ITEMS:
            tp = TaskProgress(
                recruitment_classroom_id=rc.id,
                item_key=item_key,
                item_name=item_name,
                task_type="recovery",
            )
            db.add(tp)

    db.commit()
```

```python
@app.post("/api/recruit/{recruit_id}/init-recovery")
async def init_recovery(request: Request, recruit_id: int, db: Session = Depends(get_db)):
    """开启恢复阶段：检查是否已封门，初始化恢复任务"""
    check_admin_login(request)
    check_csrf(request)

    recruit = db.query(Recruitment).filter(Recruitment.id == recruit_id).first()
    if not recruit:
        raise HTTPException(404, "招募不存在")

    # 检查是否所有教室已封门
    records = db.query(AcceptanceRecord).join(
        RecruitmentClassroom,
        AcceptanceRecord.recruitment_classroom_id == RecruitmentClassroom.id
    ).filter(RecruitmentClassroom.recruitment_id == recruit_id).all()

    not_sealed = [r for r in records if r.status != "sealed"]
    if not_sealed:
        raise HTTPException(400, f"还有 {len(not_sealed)} 间教室未封门，不能开启恢复")

    init_recovery_tasks(recruit_id, db)
    return {"code": 0, "msg": "恢复任务已创建"}


@app.post("/api/my-recovery-tasks")
async def get_my_recovery_tasks(
    student_id: str = Form(...),
    phone: str = Form(...),
    db: Session = Depends(get_db)
):
    """学生查看自己的恢复任务"""
    if not (student_id.isdigit() and len(student_id) == 8):
        raise HTTPException(400, "学号格式错误")
    if not (phone.isdigit() and len(phone) == 11):
        raise HTTPException(400, "手机号格式错误")

    reg = db.query(Registration).filter(
        Registration.student_id == student_id,
        Registration.phone == phone
    ).first()
    if not reg:
        raise HTTPException(404, "未找到报名记录")

    member = db.query(RecruitmentGroupMember).join(
        RecruitmentGroup,
        RecruitmentGroupMember.group_id == RecruitmentGroup.id
    ).filter(
        RecruitmentGroupMember.registration_id == reg.id,
        RecruitmentGroup.recruitment_id == reg.recruitment_id
    ).first()

    if not member:
        return {"code": 0, "data": []}

    assignments = db.query(RecruitmentGroupClassroom).filter(
        RecruitmentGroupClassroom.group_id == member.group_id
    ).all()

    result = []
    for assign in assignments:
        rc = db.query(RecruitmentClassroom).filter(RecruitmentClassroom.id == assign.recruitment_classroom_id).first()
        if not rc:
            continue
        cr = db.query(Classroom).filter(Classroom.id == rc.classroom_id).first()
        if not cr:
            continue

        tasks = db.query(TaskProgress).filter(
            TaskProgress.recruitment_classroom_id == rc.id,
            TaskProgress.task_type == "recovery"
        ).order_by(TaskProgress.id).all()

        if not tasks:
            continue

        task_list = [{
            "id": t.id,
            "item_key": t.item_key,
            "item_name": t.item_name,
            "is_completed": t.is_completed,
        } for t in tasks]

        completed = sum(1 for t in tasks if t.is_completed)
        total = len(tasks)

        result.append({
            "rc_id": rc.id,
            "classroom_name": cr.name,
            "tasks": task_list,
            "progress": f"{completed}/{total}",
            "all_done": completed >= total,
        })

    return {"code": 0, "data": result}


@app.get("/api/recruit/{recruit_id}/recovery-progress")
async def get_recovery_progress(request: Request, recruit_id: int, db: Session = Depends(get_db)):
    """管理员查看恢复进度"""
    check_admin_login(request)

    groups = db.query(RecruitmentGroup).filter(
        RecruitmentGroup.recruitment_id == recruit_id
    ).all()

    result = []
    for g in groups:
        assigns = db.query(RecruitmentGroupClassroom).filter(
            RecruitmentGroupClassroom.group_id == g.id
        ).all()

        for assign in assigns:
            rc = db.query(RecruitmentClassroom).filter(RecruitmentClassroom.id == assign.recruitment_classroom_id).first()
            if not rc:
                continue
            cr = db.query(Classroom).filter(Classroom.id == rc.classroom_id).first()
            if not cr:
                continue

            tasks = db.query(TaskProgress).filter(
                TaskProgress.recruitment_classroom_id == rc.id,
                TaskProgress.task_type == "recovery"
            ).all()

            completed = sum(1 for t in tasks if t.is_completed)
            total = len(tasks)

            result.append({
                "classroom_name": cr.name,
                "zone_name": g.zone_name or "无分区",
                "progress": f"{completed}/{total}",
                "percent": int(completed / total * 100) if total > 0 else 0,
                "all_done": completed >= total,
            })

    return {"code": 0, "data": result}
```

```javascript
// ========== 考后恢复 ==========

async function loadRecoveryTasks(studentId, phone) {
    const card = document.getElementById('recoveryCard');
    const container = document.getElementById('recoveryList');
    const formData = new FormData();
    formData.append('student_id', studentId);
    formData.append('phone', phone);

    try {
        const res = await fetch('/api/my-recovery-tasks', { method: 'POST', body: formData });
        const data = await res.json();

        if (data.code !== 0 || !data.data || data.data.length === 0) {
            card.style.display = 'none';
            return;
        }

        card.style.display = 'block';
        let html = '';

        data.data.forEach(item => {
            html += `<div class="card border mb-3">
                <div class="card-header d-flex justify-content-between align-items-center py-2">
                    <span class="fw-bold">🏫 ${item.classroom_name}</span>
                    <span class="badge ${item.all_done ? 'bg-success' : 'bg-warning text-dark'}">${item.progress}</span>
                </div>
                <div class="card-body py-2">`;

            item.tasks.forEach(t => {
                const checked = t.is_completed ? 'checked' : '';
                const deco = t.is_completed ? 'style="text-decoration:line-through"' : '';
                html += `<div class="form-check mb-1">
                    <input class="form-check-input recovery-checkbox" type="checkbox" data-task-id="${t.id}" ${checked}>
                    <label class="form-check-label small" ${deco}>${t.item_name}</label>
                </div>`;
            });

            html += `</div></div>`;
        });

        container.innerHTML = html;

        container.querySelectorAll('.recovery-checkbox').forEach(cb => {
            cb.addEventListener('change', async function() {
                const taskId = this.dataset.taskId;
                const label = this.nextElementSibling;
                try {
                    const res = await fetch(`/api/tasks/${taskId}/toggle`, { method: 'POST' });
                    const data = await res.json();
                    if (data.code === 0) {
                        label.style.textDecoration = data.is_completed ? 'line-through' : 'none';
                        setTimeout(() => loadRecoveryTasks(studentId, phone), 500);
                    }
                } catch (err) {
                    this.checked = !this.checked;
                    showToast('操作失败', 'danger');
                }
            });
        });
    } catch (err) {
        card.style.display = 'none';
    }
}
```

```javascript
// ========== 考后恢复 ==========

let currentRecoveryRecruitId = null;

async function openRecoveryPanel(recruitId, examName) {
    currentRecoveryRecruitId = recruitId;
    document.getElementById('recoveryModalTitle').textContent = `考后恢复 — ${examName}`;
    document.getElementById('recoveryModalBody').innerHTML = '<div class="text-center py-4 text-muted">加载中...</div>';
    bootstrap.Modal.getOrCreateInstance(document.getElementById('recoveryModal')).show();
    await loadRecoveryProgress(recruitId);
}

async function loadRecoveryProgress(recruitId) {
    try {
        const res = await fetch(`/api/recruit/${recruitId}/recovery-progress`);
        if (!res.ok) throw new Error('加载失败');
        const data = await res.json();
        renderRecovery(data.data || []);
    } catch (err) {
        document.getElementById('recoveryModalBody').innerHTML = '<div class="text-danger text-center py-4">加载失败</div>';
    }
}

function renderRecovery(items) {
    const container = document.getElementById('recoveryModalBody');
    if (!items || items.length === 0) {
        container.innerHTML = '<div class="text-center py-4"><div class="text-muted mb-3">暂无恢复任务，点击「开启恢复阶段」创建</div></div>';
        return;
    }

    const zones = {};
    items.forEach(item => {
        const z = item.zone_name || '无分区';
        if (!zones[z]) zones[z] = [];
        zones[z].push(item);
    });

    let html = '';
    let totalDone = 0;
    for (const [zone, zoneItems] of Object.entries(zones)) {
        html += `<div class="mb-3"><h5 class="border-bottom pb-1">🏢 ${zone}</h5>`;
        zoneItems.forEach(item => {
            if (item.all_done) totalDone++;
            html += `<div class="d-flex justify-content-between align-items-center border-bottom py-2">
                <strong>${item.classroom_name}</strong>
                <div class="d-flex align-items-center gap-2">
                    <div class="progress" style="width:100px;height:8px;">
                        <div class="progress-bar ${item.all_done ? 'bg-success' : 'bg-warning'}" style="width:${item.percent}%"></div>
                    </div>
                    <span class="small text-muted">${item.progress}</span>
                </div>
            </div>`;
        });
        html += '</div>';
    }
    html += `<div class="text-muted small mt-2">共 ${items.length} 间教室，已完成 ${totalDone} 间</div>`;
    container.innerHTML = html;
}

document.getElementById('initRecoveryBtn')?.addEventListener('click', async () => {
    if (!currentRecoveryRecruitId) return;
    if (!await showConfirm('确认开启', '确定开启恢复阶段吗？所有已封门教室的恢复任务将被创建。')) return;
    const csrf = document.cookie.split('; ').find(row => row.startsWith('kaowu_csrf='))?.split('=')[1] || '';
    try {
        const res = await fetch(`/api/recruit/${currentRecoveryRecruitId}/init-recovery`, {
            method: 'POST',
            headers: { 'X-CSRF-Token': csrf }
        });
        const data = await res.json();
        showToast(data.msg || data.detail, data.code === 0 ? 'success' : 'danger');
        if (data.code === 0) await loadRecoveryProgress(currentRecoveryRecruitId);
    } catch (err) {
        showToast('操作失败', 'danger');
    }
});
```
