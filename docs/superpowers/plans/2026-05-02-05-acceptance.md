# 检查验收 — 实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 布置任务全部完成后，小组提交验收，楼栋负责人检查（如有），总负责人确认封门。验收流自适应是否有楼栋负责人。

**Architecture:** 新增 AcceptanceRecord 模型记录每间教室的验收状态；学生端对楼栋负责人显示该栋所有教室的验收面板；管理员端显示全局验收总览。

**自适应逻辑：**
- 有楼栋负责人 → 自检→交叉检→**楼栋验收**→总负责人确认→封门
- 仅总负责人 → 自检→交叉检→**总负责人验收**→封门

@app.post("/api/recruit/{recruit_id}/classrooms/{rc_id}/submit-review")
async def submit_for_review(
    request: Request,
    recruit_id: int,
    rc_id: int,
    db: Session = Depends(get_db)
):
    """小组提交验收（需学号+手机验证）"""
    import json
    body = await request.json()
    student_id = body.get("student_id", "")
    phone = body.get("phone", "")

    if not (student_id.isdigit() and len(student_id) == 8):
        raise HTTPException(400, "学号格式错误")
    if not (phone.isdigit() and len(phone) == 11):
        raise HTTPException(400, "手机号格式错误")

    reg = db.query(Registration).filter(
        Registration.student_id == student_id,
        Registration.phone == phone,
        Registration.recruitment_id == recruit_id
    ).first()
    if not reg:
        raise HTTPException(404, "未找到报名记录")

    # 验证此人属于该考场所在组
    rc = db.query(RecruitmentClassroom).filter(RecruitmentClassroom.id == rc_id).first()
    if not rc:
        raise HTTPException(404, "考场不存在")

    assign = db.query(RecruitmentGroupClassroom).join(
        RecruitmentGroup, RecruitmentGroupClassroom.group_id == RecruitmentGroup.id
    ).filter(
        RecruitmentGroupClassroom.recruitment_classroom_id == rc_id,
        RecruitmentGroup.recruitment_id == recruit_id
    ).first()
    if not assign:
        raise HTTPException(403, "你没有此考场的权限")

    member = db.query(RecruitmentGroupMember).filter(
        RecruitmentGroupMember.group_id == assign.group_id,
        RecruitmentGroupMember.registration_id == reg.id
    ).first()
    if not member:
        raise HTTPException(403, "你不是该考场所在组的成员")

    # 检查任务是否全部完成
    tasks = db.query(TaskProgress).filter(
        TaskProgress.recruitment_classroom_id == rc_id,
        TaskProgress.task_type == "setup"
    ).all()
    completed = sum(1 for t in tasks if t.is_completed)
    total = sum(1 for t in tasks if not t.is_auto_skip)
    if completed < total:
        raise HTTPException(400, f"还有 {total - completed} 项任务未完成")

    # 更新验收记录
    record = db.query(AcceptanceRecord).filter(
        AcceptanceRecord.recruitment_classroom_id == rc_id
    ).first()
    if not record:
        record = AcceptanceRecord(recruitment_classroom_id=rc_id)
        db.add(record)

    if record.status == "passed":
        raise HTTPException(400, "该考场已验收通过")

    record.status = "pending"
    record.note = None
    db.commit()
    return {"code": 0, "msg": "已提交验收"}


@app.get("/api/recruit/{recruit_id}/acceptance/supervisor")
async def supervisor_acceptance_panel(
    recruit_id: int,
    student_id: str = "",
    phone: str = "",
    db: Session = Depends(get_db)
):
    """楼栋负责人查看自己楼栋的验收面板"""
    if not (student_id.isdigit() and len(student_id) == 8):
        raise HTTPException(400, "学号格式错误")
    if not (phone.isdigit() and len(phone) == 11):
        raise HTTPException(400, "手机号格式错误")

    reg = db.query(Registration).filter(
        Registration.student_id == student_id,
        Registration.phone == phone,
        Registration.recruitment_id == recruit_id
    ).first()
    if not reg:
        raise HTTPException(404, "未找到报名记录")

    # 找该人是否为楼栋负责人
    group = db.query(RecruitmentGroup).filter(
        RecruitmentGroup.recruitment_id == recruit_id,
        RecruitmentGroup.is_supervisor == True
    ).join(
        RecruitmentGroupMember,
        RecruitmentGroupMember.group_id == RecruitmentGroup.id
    ).filter(
        RecruitmentGroupMember.registration_id == reg.id
    ).first()

    if not group:
        raise HTTPException(403, "你不是楼栋负责人")

    # 查询该栋所有考场的验收状态
    zone_groups = db.query(RecruitmentGroup).filter(
        RecruitmentGroup.recruitment_id == recruit_id,
        RecruitmentGroup.zone_name == group.zone_name
    ).all()

    zone_group_ids = [g.id for g in zone_groups]
    assigns = db.query(RecruitmentGroupClassroom).filter(
        RecruitmentGroupClassroom.group_id.in_(zone_group_ids)
    ).all()

    result = []
    for assign in assigns:
        rc = db.query(RecruitmentClassroom).filter(RecruitmentClassroom.id == assign.recruitment_classroom_id).first()
        if not rc:
            continue
        cr = db.query(Classroom).filter(Classroom.id == rc.classroom_id).first()
        if not cr:
            continue

        record = db.query(AcceptanceRecord).filter(
            AcceptanceRecord.recruitment_classroom_id == rc.id
        ).first()

        # 查该组有哪些人
        members = db.query(Registration).join(
            RecruitmentGroupMember,
            RecruitmentGroupMember.registration_id == Registration.id
        ).filter(RecruitmentGroupMember.group_id == assign.group_id).all()

        result.append({
            "rc_id": rc.id,
            "classroom_name": cr.name,
            "group_members": [{"name": m.name, "gender": m.gender} for m in members],
            "status": record.status if record else "pending",
            "note": record.note if record else "",
        })

    return {"code": 0, "data": result, "zone_name": group.zone_name}


@app.post("/api/acceptance/{rc_id}/review")
async def review_classroom(
    request: Request,
    rc_id: int,
    db: Session = Depends(get_db)
):
    """楼栋负责人/管理员验收或驳回"""
    import json
    body = await request.json()
    action = body.get("action", "")       # "pass" or "reject"
    note = body.get("note", "")
    student_id = body.get("student_id", "")
    phone = body.get("phone", "")         # 验证身份

    if action not in ("pass", "reject"):
        raise HTTPException(400, "操作无效")

    if action == "reject" and not note.strip():
        raise HTTPException(400, "驳回必须填写原因")

    record = db.query(AcceptanceRecord).filter(
        AcceptanceRecord.recruitment_classroom_id == rc_id
    ).first()
    if not record:
        raise HTTPException(404, "验收记录不存在")

    if record.status == "sealed":
        raise HTTPException(400, "已封门，不可操作")

    record.status = "rejected" if action == "reject" else "passed"
    record.note = note.strip() if note.strip() else None
    db.commit()
    return {"code": 0, "msg": "操作成功"}


@app.post("/api/recruit/{recruit_id}/acceptance/seal")
async def seal_all(request: Request, recruit_id: int, db: Session = Depends(get_db)):
    """总负责人确认封门"""
    check_admin_login(request)
    check_csrf(request)

    # 检查是否所有教室都已通过
    records = db.query(AcceptanceRecord).join(
        RecruitmentClassroom,
        AcceptanceRecord.recruitment_classroom_id == RecruitmentClassroom.id
    ).filter(
        RecruitmentClassroom.recruitment_id == recruit_id
    ).all()

    not_passed = [r for r in records if r.status != "passed" and r.status != "sealed"]
    if not_passed:
        raise HTTPException(400, f"还有 {len(not_passed)} 间教室未通过验收")

    for r in records:
        r.status = "sealed"

    db.commit()
    return {"code": 0, "msg": "全部封门完成"}

async function loadSupervisorPanel(studentId, phone) {
    const card = document.getElementById('supervisorCard');
    const list = document.getElementById('supervisorList');
    const zoneLabel = document.getElementById('supervisorZone');

    // 需要从 myRegForm 获取 recruit_id — 从查询结果推断

    const regRes = await fetch('/api/my-registrations', {
        method: 'POST',
        body: new URLSearchParams({ student_id: studentId, phone: phone })
    });
    const regData = await regRes.json();
    if (regData.code !== 0 || !regData.data || regData.data.length === 0) {
        card.style.display = 'none';
        return;
    }

    // 尝试每个报名记录加载验收面板
    for (const reg of regData.data) {
        try {
            const res = await fetch(`/api/recruit/${reg.recruit_id}/acceptance/supervisor?student_id=${studentId}&phone=${phone}`);
            if (!res.ok) continue;
            const data = await res.json();
            if (data.code === 0 && data.data && data.data.length > 0) {

                        'sealed': '🔒 已封门',

                    const statusText = statusMap[item.status] || '⏳ 待验收';
                    const statusColor = item.status === 'passed' || item.status === 'sealed' ? 'success' :
                                        item.status === 'rejected' ? 'danger' : 'warning';

                        <div class="border rounded p-3">

// ========== 验收总览 ==========

let currentAcceptRecruitId = null;

async function openAcceptancePanel(recruitId, examName) {
    currentAcceptRecruitId = recruitId;
    document.getElementById('acceptanceModalTitle').textContent = `验收总览 — ${examName}`;
    document.getElementById('acceptanceModalBody').innerHTML = '<div class="text-center py-4 text-muted">加载中...</div>';
    bootstrap.Modal.getOrCreateInstance(document.getElementById('acceptanceModal')).show();
    await loadAcceptanceData(recruitId);
}

async function loadAcceptanceData(recruitId) {
    try {
        const res = await fetch(`/api/recruit/${recruitId}/acceptance/overview`);
        if (!res.ok) throw new Error('加载失败');
        const data = await res.json();
        renderAcceptance(data.data || []);
    } catch (err) {
        document.getElementById('acceptanceModalBody').innerHTML = '<div class="text-danger text-center py-4">加载失败</div>';
    }
}

function renderAcceptance(items) {
    const container = document.getElementById('acceptanceModalBody');
    if (!items || items.length === 0) {
        container.innerHTML = '<div class="text-center py-4 text-muted">暂无数据</div>';
        return;
    }

    // 按 zone 分组
    const zones = {};
    items.forEach(item => {
        const z = item.zone_name || '无分区';
        if (!zones[z]) zones[z] = [];
        zones[z].push(item);
    });

    const statusMap = {
        'pending': { label: '待验收', class: 'bg-warning text-dark' },
        'rejected': { label: '已驳回', class: 'bg-danger' },
        'passed': { label: '已通过', class: 'bg-success' },
        'sealed': { label: '已封门', class: 'bg-secondary' },
    };

    let html = '';
    let totalPassed = 0;
    for (const [zone, zoneItems] of Object.entries(zones)) {
        html += `<div class="mb-3"><h5 class="border-bottom pb-1">🏢 ${zone}</h5>`;
        zoneItems.forEach(item => {
            const st = statusMap[item.status] || statusMap['pending'];
            if (item.status === 'passed') totalPassed++;
            html += `<div class="d-flex justify-content-between align-items-center border-bottom py-2">
                <div>
                    <strong>${item.classroom_name}</strong>
                    <span class="small text-muted ms-2">👤 ${item.members.join('、')}</span>
                    ${item.note ? `<div class="small text-danger">❌ ${item.note}</div>` : ''}
                </div>
                <span class="badge ${st.class}">${st.label}</span>
            </div>`;
        });
        html += '</div>';
    }
    html += `<div class="text-muted small mt-2">共 ${items.length} 间教室，已通过 ${totalPassed} 间</div>`;

    container.innerHTML = html;
}

document.getElementById('sealAllBtn')?.addEventListener('click', async () => {
    if (!currentAcceptRecruitId) return;
    if (!await showConfirm('确认封门', '确定封门吗？所有已通过的教室将被锁定，未通过的教室保持原状态。')) return;
    const csrf = document.cookie.split('; ').find(row => row.startsWith('kaowu_csrf='))?.split('=')[1] || '';
    try {
        const res = await fetch(`/api/recruit/${currentAcceptRecruitId}/acceptance/seal`, {
            method: 'POST',
            headers: { 'X-CSRF-Token': csrf }
        });
        const data = await res.json();
        showToast(data.msg || data.detail, data.code === 0 ? 'success' : 'danger');
        if (data.code === 0) await loadAcceptanceData(currentAcceptRecruitId);
    } catch (err) {
        showToast('操作失败', 'danger');
    }
});
