    group_id = Column(Integer, nullable=False)

- [ ] **Step 1: 添加自动分组函数 + API**

在工具函数区（detect_zone 附近）添加分组算法：

```python
def auto_assign_groups(recruit_id: int, db: Session) -> list[dict]:
    """自动分组算法：按男女搭配 > 经验搭配，分栋分配教室"""
    from collections import defaultdict

    # 获取所有报名人员
    registrations = db.query(Registration).filter(
        Registration.recruitment_id == recruit_id
    ).all()

    if not registrations:
        raise HTTPException(400, "暂无报名记录")

    # 获取所有考场配置
    rcs = db.query(RecruitmentClassroom).filter(
        RecruitmentClassroom.recruitment_id == recruit_id
    ).all()

    if not rcs:
        raise HTTPException(400, "未配置考场，请先选择考场场地")

    # 获取教室信息
    classroom_ids = [rc.classroom_id for rc in rcs]
    classrooms = {c.id: c for c in db.query(Classroom).filter(Classroom.id.in_(classroom_ids)).all()}

    # 按栋分组考场
    zone_classrooms = defaultdict(list)  # {zone_name: [(rc, classroom), ...]}
    zone_no_zone = []
    for rc in rcs:
        cr = classrooms.get(rc.classroom_id)
        if not cr:
            continue
        zone = detect_zone(cr.name)
        if zone:
            zone_classrooms[zone].append((rc, cr))
        else:
            zone_no_zone.append((rc, cr))

    # 如果没有带字母的栋（综合楼的情况），把无分区的作为一组
    if not zone_classrooms and zone_no_zone:
        zone_classrooms["综合楼"].extend(zone_no_zone)
    elif zone_no_zone:
        zone_classrooms["综合楼"].extend(zone_no_zone)

    # 计算每个zone需要的总人数（一个考场一组，一组2人，双考场=2个考场）
    # 简化：每个zone需要 ceiling(考场数 / 每组的考场数) * 2 人
    # 这里设定固定桌椅教室每组可负责3间，活动桌椅每组负责2间
    zone_needs = {}  # {zone_name: (min_people, max_rooms_per_group)}
    for zone, items in zone_classrooms.items():
        total_rooms = sum(
            2 if rc.exam_mode == "double" else 1
            for rc, _ in items
        )
        # 检查该zone下的教室类型
        all_fixed = all(cr.is_fixed_seats for _, cr in items)
        rooms_per_group = 3 if all_fixed else 2
        num_groups = max(1, (total_rooms + rooms_per_group - 1) // rooms_per_group)
        zone_needs[zone] = {
            "total_rooms": total_rooms,
            "rooms_per_group": rooms_per_group,
            "num_groups": num_groups,
            "min_people": num_groups * 2,
            "classrooms": items,
        }

    # 按性别+经验排序报名人员
    # 先按性别分组：男一列，女一列
    males = [r for r in registrations if r.gender == "男"]
    females = [r for r in registrations if r.gender == "女"]

    # 同性别内：有经验优先排在前
    males.sort(key=lambda r: (0 if r.has_experience else 1))
    females.sort(key=lambda r: (0 if r.has_experience else 1))

    # 配对：男女搭配优先，剩余同性别按经验搭配
    paired_groups = []
    while males and females:
        m = males.pop(0)
        f = females.pop(0)
        paired_groups.append([m, f])
    # 处理剩余同性别
    remaining = males + females
    # 按经验搭配
    exp = [r for r in remaining if r.has_experience]
    noexp = [r for r in remaining if not r.has_experience]
    while exp and noexp:
        paired_groups.append([exp.pop(0), noexp.pop(0)])
    # 剩下的凑一起
    leftover = exp + noexp
    while leftover:
        if len(leftover) >= 2:
            paired_groups.append([leftover.pop(0), leftover.pop(0)])
        else:
            paired_groups.append([leftover.pop(0)])
            break

    # 按zone分配人员
    zones_list = list(zone_needs.keys())
    # 按需要的组数从大到小排序，优先分配更多人
    zones_list.sort(key=lambda z: zone_needs[z]["num_groups"], reverse=True)

    result = []
    group_idx = 0

    for zone in zones_list:
        info = zone_needs[zone]
        zone_groups = []

        # 为该zone分配人员组
        for _ in range(info["num_groups"]):
            if group_idx >= len(paired_groups):
                break
            group_people = paired_groups[group_idx]
            group_idx += 1

            # 推荐楼栋负责人：取组内有经验者，如果没有则取第一个
            supervisor = None
            for p in group_people:
                if p.has_experience:
                    supervisor = p
                    break
            if not supervisor and group_people:
                supervisor = group_people[0]

            zone_groups.append({
                "members": group_people,
                "supervisor": supervisor,
            })

        result.append({
            "zone": zone,
            "groups": zone_groups,
            "classrooms": info["classrooms"],
            "rooms_per_group": info["rooms_per_group"],
        })

    return result


def save_groups_to_db(recruit_id: int, db: Session) -> dict:
    """将自动分组结果写入数据库"""
    # 先删旧分组
    existing_groups = db.query(RecruitmentGroup).filter(
        RecruitmentGroup.recruitment_id == recruit_id
    ).all()
    for g in existing_groups:
        db.query(RecruitmentGroupMember).filter(
            RecruitmentGroupMember.group_id == g.id
        ).delete()
        db.query(RecruitmentGroupClassroom).filter(
            RecruitmentGroupClassroom.group_id == g.id
        ).delete()
    db.query(RecruitmentGroup).filter(
        RecruitmentGroup.recruitment_id == recruit_id
    ).delete()
    db.flush()

    groups_data = auto_assign_groups(recruit_id, db)
    saved_groups = []

    for zone_data in groups_data:
        zone = zone_data["zone"]
        rooms_per_group = zone_data["rooms_per_group"]
        classrooms_list = zone_data["classrooms"]
        classroom_idx = 0

        for g in zone_data["groups"]:
            supervisor = g["supervisor"]
            group = RecruitmentGroup(
                recruitment_id=recruit_id,
                zone_name=zone,
                is_supervisor=False,
            )
            db.add(group)
            db.flush()

            # 添加成员
            for p in g["members"]:
                member = RecruitmentGroupMember(
                    group_id=group.id,
                    registration_id=p.id,
                )
                db.add(member)

                # 如果是楼栋负责人，标记该组
                if supervisor and p.id == supervisor.id:
                    group.is_supervisor = True

            # 分配教室
            for _ in range(rooms_per_group):
                if classroom_idx >= len(classrooms_list):
                    break
                rc, _ = classrooms_list[classroom_idx]
                assign = RecruitmentGroupClassroom(
                    group_id=group.id,
                    recruitment_classroom_id=rc.id,
                )
                db.add(assign)
                classroom_idx += 1

            saved_groups.append(group)

    db.commit()
    return {"code": 0, "msg": "分组保存成功"}
```

@app.post("/api/recruit/{recruit_id}/auto-group")
async def auto_group(request: Request, recruit_id: int, db: Session = Depends(get_db)):
    """自动分组并保存"""
    check_admin_login(request)
    check_csrf(request)

@app.get("/api/recruit/{recruit_id}/groups")
async def get_groups(request: Request, recruit_id: int, db: Session = Depends(get_db)):
    """获取某个招募的所有分组信息"""
    check_admin_login(request)
    groups = db.query(RecruitmentGroup).filter(
        RecruitmentGroup.recruitment_id == recruit_id
    ).order_by(RecruitmentGroup.zone_name, RecruitmentGroup.id).all()

    # 批量查询成员和教室分配
    result = []
    for g in groups:
        members = db.query(Registration).join(
            RecruitmentGroupMember,
            RecruitmentGroupMember.registration_id == Registration.id
        ).filter(
            RecruitmentGroupMember.group_id == g.id
        ).all()

        classroom_assigns = db.query(RecruitmentGroupClassroom).filter(
            RecruitmentGroupClassroom.group_id == g.id
        ).all()
        rc_ids = [ca.recruitment_classroom_id for ca in classroom_assigns]

        classrooms_info = []
        for rcid in rc_ids:
            rc = db.query(RecruitmentClassroom).filter(RecruitmentClassroom.id == rcid).first()
            if rc:
                cr = db.query(Classroom).filter(Classroom.id == rc.classroom_id).first()
                classrooms_info.append({
                    "rc_id": rc.id,
                    "classroom_name": cr.name if cr else "未知",
                    "exam_mode": rc.exam_mode,
                    "exam_number_start": rc.exam_number_start,
                })

        result.append({
            "id": g.id,
            "zone_name": g.zone_name or "无分区",
            "is_supervisor": g.is_supervisor,
            "members": [{
                "id": m.id,
                "name": m.name,
                "student_id": m.student_id,
                "gender": m.gender,
                "has_experience": m.has_experience,
            } for m in members],
            "classrooms": classrooms_info,
        })

    return {"code": 0, "data": result}


@app.post("/api/recruit/{recruit_id}/save-groups")
async def save_groups(request: Request, recruit_id: int, db: Session = Depends(get_db)):
    """保存手动调整后的分组（管理员拖拽后保存）"""
    check_admin_login(request)
    check_csrf(request)

    import json
    body = await request.json()
    groups_data = body.get("groups", [])

    # 先清空旧分组
    existing_groups = db.query(RecruitmentGroup).filter(
        RecruitmentGroup.recruitment_id == recruit_id
    ).all()
    for g in existing_groups:
        db.query(RecruitmentGroupMember).filter(
            RecruitmentGroupMember.group_id == g.id
        ).delete()
        db.query(RecruitmentGroupClassroom).filter(
            RecruitmentGroupClassroom.group_id == g.id
        ).delete()
    db.query(RecruitmentGroup).filter(
        RecruitmentGroup.recruitment_id == recruit_id
    ).delete()
    db.flush()

    for gd in groups_data:
        group = RecruitmentGroup(
            recruitment_id=recruit_id,
            zone_name=gd.get("zone_name"),
            is_supervisor=gd.get("is_supervisor", False),
        )
        db.add(group)
        db.flush()

        for rid in gd.get("member_ids", []):
            db.add(RecruitmentGroupMember(group_id=group.id, registration_id=rid))

        for rcid in gd.get("classroom_rc_ids", []):
            db.add(RecruitmentGroupClassroom(group_id=group.id, recruitment_classroom_id=rcid))

    db.commit()
    return {"code": 0, "msg": "分组保存成功"}

// ========== 分组排班 ==========

let currentGroupingRecruitId = null;
let groupingModalInstance = null;

async function openGroupingPanel(recruitId, examName) {
    currentGroupingRecruitId = recruitId;
    document.getElementById('groupingModalTitle').textContent = `分组排班 — ${examName}`;
    document.getElementById('groupingModalBody').innerHTML = '<div class="text-center py-4 text-muted">加载中...</div>';

    if (!groupingModalInstance) {
        groupingModalInstance = new bootstrap.Modal(document.getElementById('groupingModal'));
    }
    groupingModalInstance.show();

    await loadGroupingData(recruitId);
}

async function loadGroupingData(recruitId) {
    try {
        const res = await fetch(`/api/recruit/${recruitId}/groups`);
        const data = await res.json();
        renderGrouping(data.data || []);
    } catch (err) {
        document.getElementById('groupingModalBody').innerHTML = '<div class="text-danger text-center py-4">加载失败</div>';
    }
}

function renderGrouping(groups) {
    const container = document.getElementById('groupingModalBody');

    if (!groups || groups.length === 0) {
        container.innerHTML = `
            <div class="text-center py-4">
                <div class="text-muted mb-3">暂无分组数据，请先点击「自动分组」</div>
                <button class="btn btn-custom" onclick="triggerAutoGroup()">🔄 自动分组</button>
            </div>`;
        return;
    }

    // 按 zone 分组
    const zones = {};
    groups.forEach(g => {
        const z = g.zone_name || '无分区';
        if (!zones[z]) zones[z] = [];
        zones[z].push(g);
    });

    let html = '';
    for (const [zone, zoneGroups] of Object.entries(zones)) {
        html += `<div class="mb-4"><h5 class="mb-2">🏢 ${zone}</h5><div class="row g-2" data-zone="${zone}">`;

        zoneGroups.forEach((g, idx) => {
            const memberNames = g.members.map(m =>
                `${m.name}(${m.gender})${m.has_experience ? '⭐' : ''}`
            ).join('、');
            const classroomNames = g.classrooms.map(c => c.classroom_name).join('、');
            const isSupervisor = g.is_supervisor ? ' (楼栋负责人⭐)' : '';

            html += `
                <div class="col-md-6 col-lg-4" data-group-id="${g.id}">
                    <div class="card border">
                        <div class="card-header py-2 d-flex justify-content-between align-items-center">
                            <span class="fw-bold small">第${idx+1}组${isSupervisor}</span>
                            <span class="badge bg-light text-dark">${g.zone_name}</span>
                        </div>
                        <div class="card-body py-2">
                            <div class="small mb-1">👤 ${memberNames}</div>
                            <div class="small text-muted">🏫 ${classroomNames || '未分配教室'}</div>
                        </div>
                    </div>
                </div>`;
        });

        html += `</div></div>`;
    }

    html += `<div class="text-muted small mt-2">💡 拖拽卡片可调整分组，⭐表示有经验者</div>`;

    container.innerHTML = html;
}

async function triggerAutoGroup() {
    if (!currentGroupingRecruitId) return;
    const csrf = document.cookie.split('; ').find(row => row.startsWith('kaowu_csrf='))?.split('=')[1] || '';
    const container = document.getElementById('groupingModalBody');
    container.innerHTML = '<div class="text-center py-4 text-muted">正在自动分组...</div>';

    try {
        const res = await fetch(`/api/recruit/${currentGroupingRecruitId}/auto-group`, {
            method: 'POST',
            headers: { 'X-CSRF-Token': csrf }
        });
        const data = await res.json();
        if (data.code === 0) {
            await loadGroupingData(currentGroupingRecruitId);
        } else {
            showToast(data.detail || '自动分组失败', 'danger');
            container.innerHTML = '<div class="text-center py-4 text-danger">自动分组失败</div>';
        }
    } catch (err) {
        showToast('自动分组失败', 'danger');
        container.innerHTML = '<div class="text-center py-4 text-danger">网络错误</div>';
    }
}

document.getElementById('autoGroupBtn')?.addEventListener('click', triggerAutoGroup);
